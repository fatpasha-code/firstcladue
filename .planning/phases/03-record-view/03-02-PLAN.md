    **Step 2: Add `saveCorrections` to `src/app/actions.ts`:**

    Add this Server Action to the existing actions file (do NOT overwrite existing actions):

    ```typescript
    export async function saveCorrections(
      recordId: string,
      corrections: unknown
    ): Promise<{ success: true } | { error: string }> {
      if (!recordId || typeof recordId !== 'string') {
        return { error: 'Некорректный ID записи' }
      }

      const parsed = ExtractedDataSchema.safeParse(corrections)
      if (!parsed.success) {
        return { error: 'Некорректные данные' }
      }

      const supabase = await createClient()
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) {
        return { error: 'Не авторизован' }
      }

      try {
        const { error } = await supabase
          .from('records')
          .update({
            user_corrections: parsed.data,
            updated_at: new Date().toISOString()
          })
          .eq('id', recordId)
          .eq('user_id', user.id)

        if (error) {
          console.error('[saveCorrections] error:', error)
          return { error: 'Не удалось сохранить изменения' }
        }

        return { success: true }
      } catch (error) {
        console.error('[saveCorrections] error:', error)
        return { error: 'Не удалось сохранить изменения' }
      }
    }
    ```

    Key points per project rules (.claude/rules/api.md):
    - Validate input first (ExtractedDataSchema.safeParse)
    - Auth check (supabase.auth.getUser)
    - RLS enforced by `.eq('user_id', user.id)` — defense in depth
    - try/catch with console.error for server logs
    - Human-readable error messages (no stack traces)
    - Import `ExtractedDataSchema` from `@/lib/ai/schemas`
    - This updates ONLY `user_corrections` column — never touches `extracted_data` (per D-14)
